Permission API

Muestra todos los permisos posibles dentro de lo que hay

let apis = [
    "accelerometer",
    "ambient-light-sensor",
    "background-fetch",
    "background-sync",
    "bluetooth",
    "camera",
    "clipboard-write",
    "device-info",
    "display-capture",
    "geolocation",
    "gyroscope",
    "magnetometer",
    "microphone",
    "midi",
    "nfc",
    "notifications",
    "persistent-storage",
    "push",
    "speaker-selection"
]

apis.forEach(api => {
    navigator.permissions.query({
        name : api
    }).then(p => console.log({
        api : api,
        p
    }))
})
